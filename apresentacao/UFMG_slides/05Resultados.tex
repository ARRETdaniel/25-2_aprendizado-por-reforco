\begin{frame}
    \frametitle{ROS 2 CARLA Bridge Innovation}

    \textbf{Traditional Approach:}
    \begin{itemize}
        \item Requires building carla\_msgs package
        \item Custom message types (CarlaEgoVehicleControl)
        \item Large Docker images, long build times
        \item Simulation-specific code
    \end{itemize}

    \vspace{0.3cm}

    \textbf{Our Approach:}
    \begin{itemize}
        \item $\checkmark$ Standard geometry\_msgs/Twist messages
        \item $\checkmark$ CARLA Twist-to-Control node handles conversion
        \item $\checkmark$ No custom package compilation
        \item $\checkmark$ Reduced Docker image: 2.3 GB smaller
        \item $\checkmark$ Build time: 15 minutes faster
        \item $\checkmark$ Platform-agnostic (works with real robots!)
    \end{itemize}

    \vspace{0.3cm}

    \textcolor{blue}{\textbf{Contribution:}} Modular architecture enabling seamless sim-to-real transfer
\end{frame}
